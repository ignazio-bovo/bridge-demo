contract StakeManager is IStakeManager, UUPSUpgradeable, OwnableUpgradeable {

    uint256 public rewardRate;

    constructor() initializer {}
 
    function initialize(
        IBridge _bridge,
        uint256 _rewardRate,
        bytes32 _daturaHotkey,
        address _stakePrecompile
    ) external Einitializer {
        bridge = _bridge;
        rewardRate = _rewardRate;
        __UUPSUpgradeable_init();
        daturaStakingHotkey = _daturaHotkey;
        address admin = bridge.getAdmin();
        __Ownable_init(admin);
        stakePrecompile = _stakePrecompile;
    }
 
    modifier onlyBridge() {
        if (msg.sender != address(bridge)) revert NotBridge();
        _;
    }

    function _callStakePrecompile(uint256 amount) internal {
        IStakingPrecompile stakingPrecompile = IStakingPrecompile(
            stakePrecompile
        );
        (bool success, ) = stakePrecompile.call{value: amount}(
            abi.encodeWithSelector(
                stakingPrecompile.addStake.selector,
                daturaStakingHotkey
            )
        );
        Iif (!success) revert StakingFailed();
        totalStaked += amount;
    }
 
    function _callUnstakePrecompile(uint256 amount) internal {
        IStakingPrecompile stakingPrecompile = IStakingPrecompile(
            stakePrecompile
        );
        (bool success, ) = stakePrecompile.call(
            abi.encodeWithSelector(
                stakingPrecompile.removeStake.selector,
                daturaStakingHotkey,
                amount
            )
        );
        if (!success) revert UnstakingFailed();
        totalStaked -= amount;
    }
 
    function stake(uint256 amount, address user) external payable onlyBridge {
        stakes[user].amount += amount;
        _callStakePrecompile(amount);
        emit TokensStaked(user, amount);
    }
 
    function unstake(uint256 amount, address user) external onlyBridge {
        Stake storage stakePosition = stakes[user];
        if (stakePosition.amount < amount) revert InsufficientStakedBalance();
 
        // Calculate rewards
        uint256 reward = stakePosition.amount *
            rewardRate *
            (block.number - stakePosition.lastBlockClaimed);
        uint256 totalAmount = amount + reward;
 
        // Update state first
        stakePosition.lastBlockClaimed = uint64(block.number);
        stakePosition.amount -= amount;
        _callUnstakePrecompile(totalAmount);
 
        // Accounting for rewords: approx 50% to user, 50% to participants
        uint256 halfReward = reward / 2;
        uint256 rewardToParticipants = reward - halfReward;
        uint256 rewardPerParticipant = rewardToParticipants /
            rewardRecipients.length();
        uint256 rewardToUser = reward -
            (rewardPerParticipant * rewardRecipients.length());
 
        // Then do transfers
        (bool successfulTransferToUser, ) = user.call{
            value: amount + rewardToUser
        }("");
        if (!successfulTransferToUser) revert UnstakingFailed();
 
        for (uint256 i = 0; i < rewardRecipients.length(); i++) {
            address participantReward = rewardRecipients.at(i);
            (bool successfulTransferToParticipant, ) = participantReward.call{
                value: rewardPerParticipant
            }("");
            if (!successfulTransferToParticipant) revert UnstakingFailed();
        }
    }
 
    function addBridgeParticipantReward(
        address rewardAddress
    ) external onlyOwner {
        if (rewardAddress == address(0)) revert InvalidAddress();
        if (rewardRecipients.contains(rewardAddress))
            revert RewardRecipientAlreadyExists();
        rewardRecipients.add(rewardAddress);
    }
 
    function removeBridgeParticipantReward(
        address rewardAddress
    ) external onlyOwner {
        if (!rewardRecipients.contains(rewardAddress))
            revert RewardRecipientNotFound();
        rewardRecipients.remove(rewardAddress);
    }
 
    function _authorizeUpgrade(address) internal override onlyOwner {}
}
